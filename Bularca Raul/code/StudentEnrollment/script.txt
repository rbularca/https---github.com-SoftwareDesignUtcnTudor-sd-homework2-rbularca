CREATE SCHEMA `studentenrollment` ;
use `studentenrollment`;
CREATE TABLE `students` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45),
  `dateOfBirth` DATE,
  `address` VARCHAR(45),
  PRIMARY KEY (`id`));
  
CREATE TABLE `courses` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45),
  `teacher` VARCHAR(45),
  `studyYear` VARCHAR(45),
  PRIMARY KEY (`id`));

 CREATE TABLE `enrollments` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `student_id` INT,
  `course_id` INT,
  PRIMARY KEY (`id`),
  CONSTRAINT `course`
    FOREIGN KEY (`course_id`)
    REFERENCES `courses` (`id`),
  CONSTRAINT `student`
    FOREIGN KEY (`student_id`)
    REFERENCES `students` (`id`));
